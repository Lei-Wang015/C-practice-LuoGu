#include <stdio.h>
#include <string.h>

int main ()
{
	char str[101];
	scanf("%s[^\n]", str);
	int maxn = 0, minn = 1;
	int nums[26] = {0};
	for(int i = 0; str[i] != '\0'; i++){
		nums[str[i] - 'a'] ++;
	}
	for(int i = 0; i < 26; i++){
		if(nums[i] > maxn){
			maxn = nums[i];
		}
		else if(nums[i] > 0 && nums[i] < minn)
		{
			minn = nums[i];
		}	
	}
	int diff = maxn - minn;
	if(isPrime(diff)){
		printf("Lucky Word\n");
		printf("%d\n", diff);
	}
	else{
		printf("No Answer\n");
		printf("0\n");
	}
	return 0;
}

int isPrime(int a){
	if(a == 0 || a == 1){
		return 0;
	}
	else if(a == 2){
		return 1;
	}
	else{
		for(int i = 2; i < a; i++){
			if(a % i == 0){
				return 0;
			}
		}
	}
	return 1;
}
