```C
#include <stdio.h>

int main()
{
	int n = 0, m = 0, k = 0;
	scanf("%d %d %d", &n, &m, &k);
	int nums1[n][m];
	int nums2[m][k];
	int result[n][k];
	for(int i = 0; i < n;i++){
		for(int j = 0; j < k;j++){
			result[i][j] = 0;
		}
	}
	for(int i = 0; i < n;i++){
		for(int j = 0; j < m;j++){
			scanf("%d", &nums1[i][j]);
		}
	}
	for(int i = 0; i < m;i++){
		for(int j = 0; j < k;j++){
			scanf("%d", &nums2[i][j]);
		}
	}
	for(int i = 0; i < n;i++){
		for(int j = 0; j < k;j++){
			for(int t = 0; t < m;t++){
				result[i][j] += nums1[i][t] * nums2[t][j];
			}
		}
	}
	for(int i = 0; i < n;i++){
		for(int j = 0; j < k;j++){
			printf("%d ", result[i][j]);
		}
		printf("\n");
	}
	return 0;
}
```
