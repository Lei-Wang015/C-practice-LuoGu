#include<stdio.h>
#include<math.h>
#include<stdlib.h>
int main ()
{
	int n = 0;
	scanf("%d", &n);
	int p[n];
	int j[n];
	int s[n];
	for(int i = 0;i < n;i++){
		scanf("%d", &p[i]);
	}
	for(int i = 0;i < n;i++){
		scanf("%d", &j[i]);
	}
	for(int i = 0;i < n;i++){
		scanf("%d", &s[i]);
	}
	int score[n];
	int sum = 0;
	for(int i = 0;i < n;i++){
		score[i] = s[i] - (abs(p[i] - j[i]));
		if(score[i] >= 0){
			sum += score[i];
		}
	}
	printf("%d\n", sum);
	return 0;
}
