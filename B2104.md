```C
#include <stdio.h>

int main()
{
	int row = 0, col = 0;
	scanf("%d %d", &row, &col);
	int nums1[row][col];
	int nums2[row][col];
	int result[row][col];
	for(int i = 0; i < row; i++){
		for(int j = 0; j < col;j++){
			scanf("%d", &nums1[i][j]);
		}
	}
	for(int i = 0; i < row; i++){
		for(int j = 0; j < col;j++){
			scanf("%d", &nums2[i][j]);
		}
	}
	for(int i = 0; i < row; i++){
		for(int j = 0; j < col;j++){
			result[i][j] = nums1[i][j] + nums2[i][j];
			printf("%d ",result[i][j]);
		}
		printf("\n");
	}
	return 0;
}
```
