```C
#include <stdio.h>
#include <math.h>
#include <stdlib.h>

int main()
{
	int num = 0;
	scanf("%d", &num);
	long long int *arr = (long long int*)malloc(sizeof(long long int) * num);
	for(int i = 0; i < num;i++){
		scanf("%lld", &arr[i]);
	}
	long long int min = arr[0];
	long long int max = arr[0];
	for(int i = 0; i < num;i++){
		if(arr[i] > max){
			max = arr[i];
		}
	}
	for(int i = 0; i < num;i++){
		if(arr[i] < min){
			min = arr[i];
		}
	}
	for(int i = 0; i < num;i++){
		printf("%lld ",llabs(arr[i] - max));
	}
	printf("\n");
	for(int i = 0; i < num;i++){
		printf("%lld ",llabs(arr[i] - min));
	}
	printf("\n");
	free(arr);
	arr = NULL;
	return 0;
}
```
- long long int型数据差的绝对值要用llabs
