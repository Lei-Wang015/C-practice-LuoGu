```C
#include <stdio.h>

int main()
{
	int row = 0, col = 0;
	scanf("%d %d", &row, &col);
	int nums[row][col];
	for(int i = 0; i < row; i++){
		for(int j = 0;j < col;j++){
			scanf("%d", &nums[i][j]);
		}
	}
	for(int j = 0;j < col;j++){
		for(int i = 0; i < row; i++){
			printf("%d ",nums[i][j]);
		}
		printf("\n");
	}
	return 0;
}
```
