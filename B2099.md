#include <stdio.h>
int main () 
{
	int num[5][5];
	for(int i = 0;i < 5;i++){
		for(int j = 0;j < 5;j++){
			scanf("%d", &num[i][j]);
		}
	}
	int row1 = 0, row2 = 0;
	scanf("%d %d", &row1, &row2);
	row1 --;
	row2 --;
	for(int j = 0;j < 5;j++){
		int temp = 0;
		temp = num[row1][j];
		num[row1][j] = num[row2][j];
		num[row2][j] = temp;
	}
	for(int i = 0;i < 5;i++){
		for(int j = 0;j < 5;j++){
			printf("%d ", num[i][j]);
		}
		printf("\n");
	}
	return 0;
}
