```C
#include <stdio.h>
#include <string.h>

int main()
{
	char str[1100];
	scanf("%[^\n]s", str);
	int num[26] = {0};
	for(int i = 0; str[i] != '\0'; i++){
		num[str[i] - 'a'] ++;
	}
	for(int i = 0; str[i] != '\0'; i++){
		if(1 == num[str[i] - 'a']){
			printf("%c\n", str[i]);
			return 0;
		}
	}

	printf("no\n");
	return 0;
}
```
- 只查找到第一个就返回程序 return 0；
- 不能遍历num[],遍历num[]是按照a-z的顺寻查找，而str[]才是按照输入顺序查找；
