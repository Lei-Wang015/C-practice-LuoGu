```C
#include <stdio.h>
#include <stdlib.h>

int main() {
	int num = 0;
	scanf("%d", &num);
	long long int *arr = (long long int*)malloc(sizeof(long long int) * num);
	for(int i = 0;i < num;i++){
		scanf("%lld", &arr[i]);
	}
	for(int i = 0;i < num;i++){
		if(arr[i] % 2 == 1){
			printf("%lld", arr[i]);
		}
	}
	printf("\n");
	for(int i = 0;i < num;i++){
		if(arr[i] % 2 == 0){
			printf("%lld", arr[i]);
		}
	}
	printf("\n");
	free(arr);
	arr = NULL;
	return 0;
}
```
