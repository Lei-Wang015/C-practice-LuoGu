#include <stdio.h>
#include <string.h>

int is_palindrome(char *s, int start, int end);

int main ()
{
	char str[100];
	scanf("%[^\n]s", str);
	int len = strlen(str);
	if(is_palindrome(str, 0, len - 1)){
		printf("yes\n");
	}
	else{
		printf("no\n");
	}
	return 0;
}

int is_palindrome(char *s, int start, int end){
	if(start > end){
		return 0;
	}
	while(start < end){
		if(s[start] != s[end]){
			return 0;
		}
		start ++;
		end --;
	}
	return 1;
}
