```C
#include <stdio.h>
#include <string.h>

int main() 
{
	char str[255];
	scanf("%[^\n]s", str);
	for(int i = 0; str[i] != '\0'; i++){
		if(str[i] == 'T'){
			printf("A");
		}
		else if(str[i] == 'A'){
			printf("T");
		}
		else if(str[i] == 'C'){
			printf("G");
		}
		else if(str[i] == 'G'){
			printf("C");
		}
	}
	printf("\n");
	return 0;
}
```
- 单个字符直接比较就行，不用strcmp
