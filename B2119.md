```C
#include <stdio.h>
#include <string.h>

int main ()
{
	char str[33];
	scanf("%s", str);
	int len = strlen(str);
	if(len > 3 && str[len - 1] == 'g' && str[len - 2] == 'n' && str[len - 3] == 'i'){
		str[len - 3] = '\0';
	}
	else if(len > 2 && str[len - 1] == 'y' && str[len - 2] == 'l'){
		str[len - 2] = '\0';
	}
	else if(len > 2 && str[len - 1] == 'r' && str[len - 2] == 'e'){
		str[len - 2] = '\0';
	}
	printf("%s\n",str);
	return 0;
}
```

- 用赋值为'\0'截断字符串
